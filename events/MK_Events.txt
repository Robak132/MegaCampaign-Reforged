		}
	}
}

# Uzurpacja tronu Niemiec
country_event = {
	id = gerform.1
	title = gerform.1.t
	desc = gerform.1.d
	
	picture = CIVIL_WAR_eventPicture
	
	major = yes

	is_triggered_only = yes

	option = {
		name = "gerform.1.a"
		
		every_province = {
			limit = { owned_by = FROM }
			remove_core = ROOT
			remove_claim = ROOT
		}
		ROOT = { set_government_rank = 1 }
		FROM = {
			create_subject = {
				subject_type = vassal
				subject = ROOT
			}
		}
		
		hidden_effect = {
			ROOT = { change_tag = GRV }
			FROM = { change_tag = GER }
		}
		
		ai_chance = {
			factor = 25
		}
	}
	option = {
		name = "gerform.1.b"
		
		declare_war_with_cb = {
			who = FROM
			casus_belli = cb_disloyal_vassal
		}
		
		every_province = {
			limit = { owned_by = ROOT }
			add_claim = FROM
		}
		hidden_effect = {
			ROOT = { change_tag = GRV }
			FROM = { change_tag = GER }
		}
		
		ai_chance = {
			factor = 75
		}
	}
}

# Bunt Kurlandii
country_event = {
	id = kurland.1
	title = kurland.1.t
	desc = kurland.1.d
	
	picture = CIVIL_WAR_eventPicture
	
	major = yes

	fire_only_once = yes

	trigger = {
		tag = LIT
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	option = { # War
		name = kurland.1.a
		
		39 = {
			spawn_rebels = {
				type = nationalist_rebels
				size = 1
				
				leader = "Stoigniew Sobieski"
				win = yes
				friend = KUR
			}
		}
		ai_chance = {
			factor = 90
		}
	}
	option = { # Vassal
		name = kurland.1.b
		tooltip = {
			vassalize = KUR
		}
		hidden_effect = {
			create_vassal = KUR
		}
		add_prestige = -10
		hidden_effect = {
			KUR = {
				change_government = monarchy
				add_government_reform = feudalism_reform
				adopt_reform_progress = ROOT
		
				define_ruler = {
					name = "Stoigniew I"
					dynasty = "Sobieski"
					religion = catholic
					culture = polish
					adm = 2
					dip = 0
					mil = 2
					age = 41
				}
				add_ruler_personality = architectural_visionary_personality
				define_heir = {
					name = "Gertruda"
					monarch_name = "Gertruda I"
					dynasty = "Sobieski"
					religion = catholic
					culture = polish
					female = yes
					claim = 100
					adm = 1
					dip = 1
					mil = 1
					birth_date = 1438.6.9
				}
				define_consort = {
					name = "Maria"
					country_of_origin = KUR
					religion = catholic
					culture = baltorussian
					female = yes
					adm = 1
					dip = 0
					mil = 1
					birth_date = 1407.1.26
				}
				add_queen_personality = industrious_personality
			}
		}
		ai_chance = {
			factor = 10
		}
	}
}
