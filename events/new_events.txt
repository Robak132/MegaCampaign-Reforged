#Upadek Szczecina
country_event = {
	id = szczecin.1
	title = szczecin.1.t
	desc = szczecin.1.d
	
	picture = HARBOUR_eventPicture
	
	trigger = {
		NOT = { exists = SZZ }
		
		any_owned_province = {
			has_province_modifier = szczecin_rep
		}
	}
	
	mean_time_to_happen = {
		months = 1
	}
	
	option = {
		name = "szczecin.1.a"
		
		every_owned_province = {
			remove_province_modifier = szczecin_rep
		}
	}
}

# Uzurpacja tronu Niemiec
country_event = {
	id = gerform.1
	title = gerform.1.t
	desc = gerform.1.d
	
	picture = CIVIL_WAR_eventPicture

	is_triggered_only = yes
		
	option = {
		name = "gerform.1.a"
		
		every_province = {
			limit = { owned_by = FROM }
			remove_core = ROOT
			remove_claim = ROOT
		}
		ROOT = { set_government_rank = 1 }
		FROM = {
			create_subject = {
				subject_type = vassal
				subject = ROOT
			}
		}
		
		hidden_effect = {
			ROOT = { change_tag = GRV }
			FROM = { change_tag = GER }
		}
		
		ai_chance = {
			factor = 25
		}
	}
	option = {
		name = "gerform.1.b"
		
		declare_war_with_cb = {
			who = FROM
			casus_belli = cb_disloyal_vassal
		}
		
		every_province = {
			limit = { owned_by = ROOT }
			add_claim = FROM
		}
		hidden_effect = {
			ROOT = { change_tag = GRV }
			FROM = { change_tag = GER }
		}
		
		ai_chance = {
			factor = 75
		}
	}
}

#Konwersja Holdingdów Muzułmańskich na Chrześcijańskie

	id = holding_conversion.1
	title = holding_conversion.1.t
	desc = holding_conversion.1.d
	
	picture = CIVIL_WAR_eventPicture
	
	hidden = yes
	
	trigger = {
			OR = {
				religion = coptic
				religion = orthodox
				religion = reformed
				religion = protestant
				religion = anglican	
				religion = catholic
			}
			
			muslim_holdings = yes
	}

		days = 1
	}
	
	option = {
		every_owned_province = {
			muslim_to_christian_conversion_effect = yes
		}
	}
}
